Без отчёта

На BR-SRV:

```
apt install ansible sshpass -y
```

Далее:

```
mkdir /etc/ansible
cat > /etc/ansible/ansible.cfg <<EOF
[defaults]
host_key_checking = False
inventory = hosts.ini
EOF
cat > /etc/ansible/hosts.ini <<EOF
hq-srv ansible_user=sshuser ansible_port=2011 ansible_password=P@ssw0rd
hq-cli ansible_user=locadm ansible_password=P@ssw0rd
hq-rtr ansible_user=net_admin ansible_password=P@ssw0rd
br-rtr ansible_user=net_admin ansible_password=P@ssw0rd
EOF
cd /etc/ansible
ansible all -m ping
```
